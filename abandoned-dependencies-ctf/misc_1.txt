import socket

def Main():
    host = 'xxx.xx.xxx.xxx' #The host on your client needs to be the external-facing IP address of your router. Obtain it from here https://www.whatismyip.com/

    port = 42424
    s = socket.socket()
    s.bind((host, port))
    print("Bound socket")

    s.listen(1)
    while True:
      c, addr = s.accept()
      print("Got connection")
      data = c.recv(1024)
      data = str(data.decode())
      print("Got a password: " + data)
      c.close()

if __name__ == '__main__':
    Main()
