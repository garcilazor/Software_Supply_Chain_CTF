import socket

def Main():
    s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    s.connect(("8.8.8.8", 12345))
    host = s.getsockname()[0]
    s.close()

    s = socket.socket()
    s.bind((host, 12345))
    print("Bound socket")

    s.listen(1)
    while True:
      c, addr = s.accept()
      print("Got connection")
      data = c.recv(1024)
      data = str(data.decode())
      print("Got a password: " + data)
      c.close()

if __name__ == '__main__':
    Main()
