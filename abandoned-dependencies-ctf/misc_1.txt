import socket

def Main():
    host = 'xxx.xx.xxx.xxx'
    s = socket.socket()
    s.bind((host, 12345))
    print("Bound socket")

    s.listen(1)
    while True:
      c, addr = s.accept()
      print("Got connection")
      data = c.recv(1024)
      data = str(data.decode())
      print("Got a password: " + data)
      c.close()

if __name__ == '__main__':
    Main()
